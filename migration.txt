curl -L https://github.com/golang-migrate/migrate/releases/download/v4.16.2/migrate.linux-amd64.tar.gz | tar xvz migrate

# PATH ga qo'shish --------------------|
sudo mv migrate /usr/local/bin/migrate


# Tekshirish --------------------|
migrate -version


# Yangi fayl yaratish -------------------- | 
migrate create -ext sql -dir resources/migrations -seq fayl_nomi

# Hammani ishga tushirish -------------------------| 
make migrate-up

# Bitta orqaga ---------------- | 
make migrate-down

# Qaysi versiondasiz --------------------- | 
migrate -path resources/migrations -database "$(DB_URL)" version

# Dirty holatni tuzatish ------------------ | 
migrate -path resources/migrations -database "$(DB_URL)" force 2
